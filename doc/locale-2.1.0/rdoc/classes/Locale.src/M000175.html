  def candidates(options = {})
    opts = {
      :supported_language_tags => nil,
      :current                 => current,
      :type                    => :common,
    }.merge(options)

    Thread.current[:candidates_caches] ||= {}
    Thread.current[:candidates_caches][opts] ||=
      collect_candidates(opts[:type], opts[:current],
                         opts[:supported_language_tags])
  end
