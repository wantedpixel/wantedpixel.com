      def initialize(two_code, three_code, scope, type, name)
        @two_code, @three_code, @scope, @type, @name = two_code, three_code, scope, type, name

        @individual   = (scope == "I")
        @macro        = (scope == "M")
        @special      = (scope == "S")
        @constructed  = (type == "C")
        @living       = (type == "L")
        @extinct      = (type == "E")
        @ancient      = (type == "A")
        @historical   = (type == "H")
        @special_type = (type == "S")
      end
