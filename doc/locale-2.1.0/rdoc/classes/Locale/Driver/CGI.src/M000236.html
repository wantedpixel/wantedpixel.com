      def charset
        req = Thread.current[:current_request]
        return nil unless req

        charsets = req[:accept_charset]
        if charsets and charsets.size > 0
          num = charsets.index(',')
          charset = num ? charsets[0, num] : charsets
          charset = nil if charset == "*"
        else
          charset = nil
        end
        charset
      end
