      def g
        @g ||= ::GraphViz.digraph(@graph_name, {:concentrate => true, :normalize => true, :nodesep => 0.55}) do |g|
          g.edge[:weight]   = 2
          g.edge[:fontname] = g.node[:fontname] = 'Arial, Helvetica, SansSerif'
          g.edge[:fontsize] = 12
        end
      end
