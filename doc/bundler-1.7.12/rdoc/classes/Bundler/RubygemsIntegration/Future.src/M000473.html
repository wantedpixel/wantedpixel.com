      def download_gem(spec, uri, path)
        require 'resolv'
        uri = Bundler::Source.mirror_for(uri)
        proxy, dns = configuration[:http_proxy], Resolv::DNS.new
        fetcher = Gem::RemoteFetcher.new(proxy, dns)
        fetcher.download(spec, uri, path)
      end
