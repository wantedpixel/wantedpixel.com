      def additional_watch_paths
        Compass.configuration.watches.map do |watch|
          pathname = Pathname.new(File.join(Compass.configuration.project_path, watch.glob))
          real_path = nil
          pathname.ascend do |p|
            if p.exist?
              real_path = p
              break
            end
          end
          real_path
        end.compact.uniq
      end
