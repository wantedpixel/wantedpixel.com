  def sass_files(options = {})
    @compiler.template_location_array.map do |(sass_dir, css_dir)|
      glob = options[:include_partials] ?
               File.join("**","*.s[ac]ss*") :
               File.join("**","[^_]*.s[ac]ss*")
      Dir.glob(File.join(sass_dir, glob))
    end.flatten
  end
