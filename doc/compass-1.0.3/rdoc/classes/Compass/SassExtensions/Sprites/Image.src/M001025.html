        def repeat
          @repeat ||= begin
            rep = (get_var_file("repeat") || options.get_var("repeat") || identifier(NO_REPEAT)).value
            unless VALID_REPEATS.include? rep
              raise SpriteException, "Invalid option for repeat \"#{rep}\" - valid options are #{VALID_REPEATS.join(', ')}"
            end

            rep
          end
        end
