    def attach(mod, mname)
      invoker = self
      params = "*args"
      call = "call"
      mod.module_eval "@@\#{mname} = invoker\ndef self.\#{mname}(\#{params})\n@@\#{mname}.\#{call}(\#{params})\nend\ndef \#{mname}(\#{params})\n@@\#{mname}.\#{call}(\#{params})\nend\n"
      invoker
    end
