  def all_prefixes(browser)
    assert_valid_browser browser
    data = browser_data(browser)
    prefixes = ["-#{data["prefix"]}"]
    if data["prefix_exceptions"]
      prefixes += data["prefix_exceptions"].values.uniq.map{|p| "-#{p}"}
    end
    prefixes
  end
