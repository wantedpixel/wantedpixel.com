  def split_filename(path)
    pathname = Pathname.new(path.value)
    list(quoted_string(pathname.dirname.to_s),
         quoted_string(pathname.basename(pathname.extname).to_s),
         quoted_string(pathname.extname.to_s),
         :space)
  end
