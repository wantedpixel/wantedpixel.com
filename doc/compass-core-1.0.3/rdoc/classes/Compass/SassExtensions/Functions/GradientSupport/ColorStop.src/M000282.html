    def to_s(options = self.options)
      s = self.class.color_to_s(color)
      if stop
        s << " "
        if stop.respond_to?(:unitless?) && stop.unitless?
          s << stop.times(number(100, "%")).inspect
        else
          s << stop.to_s
        end
      end
      s
    end
