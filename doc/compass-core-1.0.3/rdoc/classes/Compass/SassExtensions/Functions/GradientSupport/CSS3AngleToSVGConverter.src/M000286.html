    def in_radians(angle)
      case angle.unit_str
      when "deg"
        angle.value * PI / 180.0
      when "grad"
        angle.value * PI / 200.0
      when "rad"
        angle.value
      when "turn"
        angle.value * PI * 2
      else
        raise Sass::SyntaxError.new("#{angle.unit_str} is not an angle")
      end
    end
