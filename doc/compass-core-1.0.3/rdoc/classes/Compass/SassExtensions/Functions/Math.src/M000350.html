  def deprecated_random(*args)
    if args.length == 2
      Compass::Util.compass_warn "WARNING: The $start value for random(\#{args.first}, \#{args.last}) is not supported by Sass and is now\n  deprecated in Compass and will be removed in a future release.\n  Use `\#{args.first} + random(\#{args.last.minus(args.first)})` instead.\n"
      range = (args.first.value..args.last.value).to_a
      number(range[rand(range.length)])
    else
      sass_random(*args)
    end
  end
