  def browsers(prefix = nil)
    browsers = if prefix
                 assert_type prefix, :String
                 Compass::Core::CanIUse.instance.browsers_with_prefix(prefix.value)
               else
                 Compass::Core::CanIUse.instance.browsers
               end
    list(browsers.map{|b| identifier(b)}, :comma)
  end
